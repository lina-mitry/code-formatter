public static String getLexemes(String content, int i) {
int      j = i;
}
while (j < content.length()) {
            if (Character.isLetter(content.charAt(j))) {
                j++;
            } else {
                if (i == j) {
                    return String.valueOf(content.charAt(i));
                }
                return content.substring(i, j);
            }
        }
        return content.substring(i, j);
    }
